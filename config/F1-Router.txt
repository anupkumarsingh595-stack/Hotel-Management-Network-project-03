hostname F1-ROUTER
no ip domain-lookup

! ===== LAN INTERFACE =====
interface GigabitEthernet0/0
 no shutdown

interface GigabitEthernet0/0.80
 encapsulation dot1Q 80
 ip address 192.168.8.1 255.255.255.0

interface GigabitEthernet0/0.70
 encapsulation dot1Q 70
 ip address 192.168.7.1 255.255.255.0

interface GigabitEthernet0/0.60
 encapsulation dot1Q 60
 ip address 192.168.6.1 255.255.255.0

! ===== WAN INTERFACES =====
interface Serial0/2/0
 ip address 10.10.10.1 255.255.255.252
 clock rate 64000
 no shutdown

interface Serial0/2/1
 ip address 10.10.10.9 255.255.255.252
 clock rate 64000
 no shutdown

! ===== OSPF =====
router ospf 1
 network 192.168.0.0 0.0.255.255 area 0
 network 10.10.10.0 0.0.0.15 area 0

! ===== DHCP =====
ip dhcp excluded-address 192.168.6.1 192.168.6.10
ip dhcp excluded-address 192.168.7.1 192.168.7.10
ip dhcp excluded-address 192.168.8.1 192.168.8.10

ip dhcp pool VLAN60
 network 192.168.6.0 255.255.255.0
 default-router 192.168.6.1

ip dhcp pool VLAN70
 network 192.168.7.0 255.255.255.0
 default-router 192.168.7.1

ip dhcp pool VLAN80
 network 192.168.8.0 255.255.255.0
 default-router 192.168.8.1

ip domain-name hotel.local
username admin privilege 15 secret admin123
crypto key generate rsa modulus 1024
